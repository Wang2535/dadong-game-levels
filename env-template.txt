# OpenClaw Skills 环境变量配置模板
# 复制此文件为 .env，并填入你的 API 密钥

# ==========================================
# Google Gemini API
# 获取地址：https://makersuite.google.com/app/apikey
# ==========================================
GOOGLE_API_KEY=your-google-api-key-here

# ==========================================
# OpenAI API
# 获取地址：https://platform.openai.com/api-keys
# ==========================================
OPENAI_API_KEY=your-openai-api-key-here

# ==========================================
# Perplexity API
# 获取地址：https://www.perplexity.ai/settings/api
# ==========================================
PERPLEXITY_API_KEY=your-perplexity-api-key-here

# ==========================================
# Todoist API
# 获取地址：https://todoist.com/app/settings/integrations/developer
# ==========================================
TODOIST_API_KEY=your-todoist-api-key-here

# ==========================================
# Notion API
# 获取地址：https://www.notion.so/my-integrations
# ==========================================
NOTION_API_KEY=your-notion-api-key-here
