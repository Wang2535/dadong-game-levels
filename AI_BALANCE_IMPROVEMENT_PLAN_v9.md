# 第一部分：方案概述

## 1.1 项目背景

### 当前问题诊断

| 问题类别 | 具体问题 | 严重程度 | 影响范围 |
|---------|---------|---------|---------|
| **卡牌不平衡** | 攻击方47张 vs 防御方16张卡牌 | 🔴 严重 | 全局 |
| **资源系统偏向** | 防御方资源获取速度过慢 | 🔴 严重 | 全局 |
| **权限系统失衡** | 攻击方渗透等级提升过快 | 🟠 中等 | 中后期 |
| **胜利条件不对称** | 攻击方胜利条件多样且容易达成 | 🟠 中等 | 全局 |
| **持续效果滥用** | 多个持续伤害效果可无限叠加 | 🔴 严重 | 中后期 |
| **区域控制缺陷** | 控制机制不完善，效果不明显 | 🟡 轻微 | 战略层面 |
| **角色技能不平衡** | 攻击方角色技能强于防御方 | 🟠 中等 | 角色选择 |

> **术语解释**:
> - **渗透等级**: 攻击方核心资源，0-10级，达到10级触发胜利
> - **安全等级**: 防御方核心资源，0-10级，降至0级触发失败
> - **持续效果**: 在多个回合内持续生效的卡牌效果

### 改进目标

1. **平衡性目标**: 攻防双方胜率差控制在10%以内
2. **体验目标**: 单局游戏时长控制在15-25分钟
3. **多样性目标**: 每张卡牌使用率超过30%
4. **稳定性目标**: 游戏崩溃率低于1%

### 2.2.2 详细调整说明

#### ATTACK_046: APT持久潜伏 综合调整

**原配置**:
- 资源消耗: `funds: 4, information: 3, access: 1`
- 持续效果: 每回合窃取1点信息
- 持续时间: 永久

**新配置**:
- 资源消耗: `funds: 5, information: 4, access: 1`
- 持续效果: 每2回合窃取1点信息
- 持续时间: 永久（但可被清除）

**调整计算**:
```
原性价比: (4+3+1)资源 / 每回合1信息 = 8资源/信息/回合
新性价比: (5+4+1)资源 / 每2回合1信息 = 20资源/信息/回合
性价比降低: 60%
```

/**
 * 处理下回合资源恢复效果
 * @param gameState 当前游戏状态
 * @param player 目标玩家
 * @param resourceType 资源类型
 * @param amount 恢复数量
 * @returns 更新后的游戏状态
 */
function handleResourceRegen(
  gameState: GameState,
  player: Player,
  resourceType: keyof Resources,
  amount: number
): GameState {
  const nextTurnEffect = {
    type: 'resource_regen',
    playerId: player.id,
    resourceType,
    amount,
    triggerTurn: gameState.turn + 1
  };
  
  return {
    ...gameState,
    nextTurnEffects: [...(gameState.nextTurnEffects || []), nextTurnEffect]
  };
}

/**
 * 处理区域控制强度提升
 * @param gameState 当前游戏状态
 * @param area 目标区域
 * @param strength 提升强度（负值表示偏向防御方）
 * @returns 更新后的游戏状态
 */
function handleControlBoost(
  gameState: GameState,
  area: AreaType,
  strength: number
): GameState {
  const areaState = gameState.areas[area];
  const newStrength = Math.max(-5, Math.min(5, 
    (areaState.controlStrength || 0) + strength
  ));
  
  return {
    ...gameState,
    areas: {
      ...gameState.areas,
      [area]: {
        ...areaState,
        controlStrength: newStrength
      }
    }
  };
}

### 4.2.2 持续效果限制实现

```typescript
/**
 * 检查是否可以放置持续效果
 * @param gameState 当前游戏状态
 * @param area 目标区域
 * @param player 效果来源玩家
 * @returns 检查结果和原因
 */
function canPlaceContinuousEffect(
  gameState: GameState,
  area: AreaType,
  player: Player
): { allowed: boolean; reason?: string } {
  const areaState = gameState.areas[area];
  
  // 检查区域持续效果数量
  const areaEffects = areaState.tokens.filter(t => 
    t.duration > 0 && t.owner === player.id
  ).length;
  
  if (areaEffects >= 2) {
    return {
      allowed: false,
      reason: `区域${area}已有${areaEffects}个持续效果，达到上限`
    };
  }
  
  // 检查玩家全局持续效果数量
  const playerEffects = Object.values(gameState.areas).reduce(
    (count, area) => count + area.tokens.filter(t => 
      t.duration > 0 && t.owner === player.id
    ).length,
    0
  );
  
  if (playerEffects >= 3) {
    return {
      allowed: false,
      reason: `玩家${player.name}已有${playerEffects}个持续效果，达到上限`
    };
  }
  
  return { allowed: true };
}
```

/**
 * 应用游戏节奏效果
 * @param gameState 当前游戏状态
 * @returns 更新后的游戏状态
 */
function applyGamePaceEffects(gameState: GameState): GameState {
  const turn = gameState.turn;
  let updatedState = { ...gameState };
  
  // 前期（1-3回合）
  if (turn <= 3) {
    updatedState.players = updatedState.players.map(player => {
      if (player.faction === 'defense') {
        return {
          ...player,
          tempDefenseBonus: 1,
          resources: {
            ...player.resources,
            compute: player.resources.compute + 1
          }
        };
      } else {
        return {
          ...player,
          maxInfiltrationGain: 0.5,
          maxContinuousEffects: 1
        };
      }
    });
  }
  
  // 中期（4-6回合）
  else if (turn <= 6) {
    updatedState.players = updatedState.players.map(player => {
      if (player.faction === 'defense') {
        return {
          ...player,
          cardCostReduction: 1,
          tempDefenseBonus: 1
        };
      } else {
        return {
          ...player,
          continuousEffectMultiplier: 0.5
        };
      }
    });
  }
  
  // 后期（7-12回合）
  else {
    updatedState.players = updatedState.players.map(player => {
      if (player.faction === 'defense') {
        return {
          ...player,
          permissions: player.permissions ? {
            ...player.permissions,
            securityLevel: Math.min(10, player.permissions.securityLevel + 1)
          } : undefined
        };
      } else {
        return {
          ...player,
          advancedCardCostIncrease: 1
        };
      }
    });
  }
  
  return updatedState;
}
